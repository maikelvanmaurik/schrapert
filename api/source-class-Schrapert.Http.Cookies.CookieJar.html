 90:     {
 91:         if (!$domain) {
 92:             $this->cookies = [];

 94:         } elseif (!$path) {

131:     public function setCookie(SetCookie $cookie)
132:     {
133:         // If the name string is empty (but not 0), ignore the set-cookie
134:         // string entirely.
135:         $name = $cookie->getName();
136:         if (!$name && $name !== '0') {
137:             return false;
138:         }
139: 
140:         // Only allow cookies with set and valid domain, name, value
141:         $result = $cookie->validate();
142:         if ($result !== true) {
143:             if ($this->strictMode) {
144:                 throw new \RuntimeException('Invalid cookie: ' . $result);
145:             } else {
146:                 $this->removeCookieIfEmpty($cookie);
147:                 return false;
148:             }
149:         }
150: 
151:         // Resolve conflicts with previously set cookies
152:         foreach ($this->cookies as $i => $c) {
153: 
154:             // Two cookies are identical, when their path, and domain are
155:             // identical.
156:             if ($c->getPath() != $cookie->getPath() ||
157:                 $c->getDomain() != $cookie->getDomain() ||
158:                 $c->getName() != $cookie->getName()
159:             ) {
160:                 continue;
161:             }
162: 
163:             // The previously set cookie is a discard cookie and this one is
164:             // not so allow the new cookie to be set
165:             if (!$cookie->getDiscard() && $c->getDiscard()) {
166:                 unset($this->cookies[$i]);
167:                 continue;
168:             }
169: 
170:             // If the new cookie's expiration is further into the future, then
171:             // replace the old cookie
172:             if ($cookie->getExpires() > $c->getExpires()) {
173:                 unset($this->cookies[$i]);
174:                 continue;
175:             }
176: 
177:             // If the value has changed, we better change it
178:             if ($cookie->getValue() !== $c->getValue()) {
179:                 unset($this->cookies[$i]);
180:                 continue;
181:             }
182: 
183:             // The cookie exists, so no need to continue
184:             return false;
185:         }
186:

188: 
189:         return true;
190:     }
