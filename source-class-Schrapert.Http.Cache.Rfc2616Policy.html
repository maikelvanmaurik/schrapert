112:     public function shouldCacheResponse(ResponseInterface $response, RequestInterface $request)
113:     {
114:         $cc = $this->parseCacheControlHeader($response);
115:         if (array_key_exists('no-store', $cc)) {
116:             return false;
117:         }
118:         // Never cache 304 (Not Modified) responses
119:         if ($response->getStatusCode() == 304) {
120:             return false;
121:         }
122:         // Cache unconditionally if configured to do so
123:         if ($this->alwaysStore) {
124:             return true;
125:         }
126: 
127:         // Any hint on response expiration is good
128:         if (array_key_exists('max-age', $cc) || $response->hasHeader('Expires')) {
129:             return true;
130:         }
131:         // Firefox fall-backs this statuses to one year expiration if none is set
132:         if (in_array($response->getStatusCode(), [300, 301, 308])) {
133:             return true;
134:         }
135:         // Other statuses without expiration requires at least one validator
136:         if (in_array($response->getStatusCode(), [200, 203, 401])) {
137:             return $response->hasHeader('Last-Modified') || $response->hasHeader('ETag');
138:         }
139:         // Any other is probably not eligible for caching
140:         // Makes no sense to cache responses that does not contain expiration
141:         // info and can not be re-validated
142:         return false;

163:     /**
164:      * https://developer.mozilla.org/en-US/docs/Web/HTTP/Caching#Freshness describes how the
165:      * freshness lifetime is calculated
166:      *
167:      * @param ResponseInterface $response
168:      * @param RequestInterface $request
169:      * @param int $now
170:      * @return int
171:      */
172:     private function computeFreshnessLifetime(ResponseInterface $response, RequestInterface $request, $now)
173:     {
174:         $cc = $this->parseCacheControlHeader($response);
175:         if (null !== ($maxAge = $this->getMaxAge($cc))) {
176:             return $maxAge;
177:         }
178: 
179:         if (null === ($date = $this->rfc1123ToEpoch($response->getHeaderLine('Date')))) {
180:             $date = $now;
181:         }
182: 
183:         if ($response->hasHeader('Expires')) {
184:             // When parsing Expires header fails RFC 2616 section 14.21 says we
185:             // should treat this as an expiration time in the past.
186:             if (null === ($expires = $this->rfc1123ToEpoch($response->getHeaderLine('Expires')))) {
187:                 return 0;
188:             }
189:             return max(0, $expires - $date);
190:         }
191: 
192:         // Fallback to heuristic using last-modified header
193:         // This is not in RFC but on Firefox caching implementation
194:         $lastModified = $this->rfc1123ToEpoch($response->getHeaderLine('Last-Modified'));
195:         if (null !== $lastModified && $lastModified <= $date) {
196:             return floor(($date - $lastModified) / 10);
197:         }
198: 
199:         // This request can be cached indefinitely
200:         if (in_array($response->getStatusCode(), [300, 301, 308])) {
201:             return self::MAX_AGE;
202:         }
203: 
204:         // Insufficient information to compute freshness lifetime
205:         return 0;
206:     }

223:     public function isCachedResponseFresh(ResponseInterface $cachedResponse, RequestInterface $request)
224:     {
225:         $ccResponse = $this->parseCacheControlHeader($cachedResponse);
226:         $ccRequest = $this->parseCacheControlHeader($request);
227: 
228:         if (array_key_exists('no-cache', $ccResponse) || array_key_exists('no-cache', $ccRequest)) {
229:             return false;
230:         }
231: 
232:         $now = $this->getCurrentTime();
233:         $freshnessLifetime = $this->computeFreshnessLifetime($cachedResponse, $request, $now);
234:         $currentAge = $this->computeCurrentAge($cachedResponse, $request, $now);
235: 
236:         if(null !== ($requestMaxAge = $this->getMaxAge($ccRequest))) {
237:             if(0 === $freshnessLifetime) { // When there was to little data to compute the freshness lifetime by the response just use the request max age
238:                 $freshnessLifetime = $requestMaxAge;
239:             } else {
240:                 $freshnessLifetime = min($freshnessLifetime, $requestMaxAge);
241:             }
242:         }
243: 
244:         if(array_key_exists('min-fresh', $ccRequest)) {
245:             $minFreshSeconds = $ccRequest['min-fresh'];
246:             if($currentAge + $minFreshSeconds > $freshnessLifetime) {
247:                 return false;
248:             }
249:         }
250: 
251:         if($currentAge < $freshnessLifetime) {
252:             return true;
253:         }
254: 
255:         if(array_key_exists('max-stale', $ccRequest) && !array_key_exists('must-revalidate', $ccResponse)) {
256:             # From RFC2616: "Indicates that the client is willing to
257:             # accept a response that has exceeded its expiration time.
258:             # If max-stale is assigned a value, then the client is
259:             # willing to accept a response that has exceeded its
260:             # expiration time by no more than the specified number of
261:             # seconds. If no value is assigned to max-stale, then the
262:             # client is willing to accept a stale response of any age."
263:             $staleAge = $ccRequest['max-stale'];
264: 
265:             if (null === $staleAge) {
266:                 return true;
267:             }
268: 
269:             if($currentAge < ($freshnessLifetime + max(0, intval($staleAge)))) {
270:                 return true;
271:             }
272:         }
273: 
274:         return false;
275:     }
